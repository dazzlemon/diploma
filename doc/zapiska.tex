\newenvironment{centerPage}[1]{
	\pagebreak
	\hspace{0pt}
	\vfill
	\begin{center}
		\Large\textbf{#1}

}

\newenvironment{wip}[1]{
	\AddToShipoutPicture{
		\begin{tikzpicture}[overlay]
			\node[opacity=0.2, rotate=45, scale=15] at (current page.center) {WIP};
		\end{tikzpicture}
	}

	#1

	\ClearShipoutPicture
}

\newenvironment{wipPage}[1]{
	\wip{
		\centerPage{#1}
	}
}
